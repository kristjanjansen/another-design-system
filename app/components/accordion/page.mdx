# Accordion and AccordionGroup

## Import

```jsx
import { Accordion, AccordionGroup } from '@enefit/eds'
```

## Single mode

```jsx
<AccordionGroup>
    <Accordion id="first" title="First">
        Lorem ipsum dolor, sit amet consectetur adipisicing elit.
    </Accordion>
    <Accordion id="second" title="Second">
        Lorem ipsum dolor, sit amet consectetur adipisicing elit.
    </Accordion>
</AccordionGroup>
```

## Multiple mode

```jsx
<AccordionGroup mode="multiple">
    <Accordion id="first" title="First">
        Lorem ipsum dolor, sit amet consectetur adipisicing elit.
    </Accordion>
    <Accordion id="second" title="Second">
        Lorem ipsum dolor, sit amet consectetur adipisicing elit.
    </Accordion>
</AccordionGroup>
```
